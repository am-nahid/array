Unique Color Shirt

function Unique_Shirts (arr,N) {
        let freq=[];//creating empty array as freq
        for(i=0;i< 1001;i++){
          freq[i]=0;
        }
        
        for(j=0;j< N;j++){
          freq[arr[j]]++;
        } 
        let count=0;
        for(k=0;k< 1001;k++){
          if(freq[k]===1){
          count++;
          }
        }
        return count;
       }

Min and Max

 
        function arrayMin(arr) {
     
       let result = arr[0];
       for (let i=0; i < arr.length; i++) {
           if (arr[i] < result) {
               result = arr[i];
           }
       }
       
       return(result);
       }
       
       
       function arrayMax(arr) {
        
       let result = arr[0];
       for (let i=0; i < arr.length; i++) {
           if (arr[i] > result) {
               result = arr[i];
           }
       }
       
       return(result);
       }

Birthday Game

function Birthday_Game(arr,D ,M) {
        N=arr.length;
        var count=0;
        for (i=0;i< N-M+1;i++){
          let sum=0;
          for (j=i;j< M+i;j++){
            sum+=arr[j];
          }
          if (sum===D){
            count++;
          }
        }
        return count;
